 public class User {

     private String password;
     private String email;
     private int age;
     private int groupId;

     }

         this.password = password;
         this.email = email;
         this.age = age;
         this.groupId = groupId;
     }

         this.username = username;
     }

         this.password = password;
     }
     @Column(name = "email")

         return email;
     }

         this.email = email;
     }
     @Column(name = "age")
     public int getAge() {
